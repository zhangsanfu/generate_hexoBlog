### 真正意义上的面向对象
> 简易版
```
class Person{
    constructor(name,age){  //构造函数
        this.name=name;
        this.age=age;
    }
   showName(){
       return this.name;
   }
}

var p1=new Person('bbb',40);
alert(p1.showName());
```
> 继承
```
class Person{
    constructor(name,age){  //构造函数
        this.name=name;
        this.age=age;
    }
    showName(){
        return this.name;
    }
}
        //继承
class Worker extends Person{
    constructor(name,age,job){
        super(name,age);
        this.job=job;
    }
    showJob(){
        return this.job;
    }
}

var p1=new Person('bbb',40);
var w1=new Worker('aaa',10,'学生');

alert(p1.showJob);
//alert(w1.showJob());
```
