###  扩展运算符  => ...
```
// var arr=[1,2,3,4,5];
// var arr2=arr;
// arr2.pop();

// alert(arr);

function show(...arr){
    console.log(arr)
}
show([1,2,3,4])  //  Array(1)
```
```
function show(...a){ 
    console.log(a.length);  //4
}
 show(1,2,3,4)
```

### 快速复制一个数组 
> ES6以前
```
var arr=[1,2,3,4,5];
var arr2=[];

for(var i=0; i<arr.length; i++){
    arr2.push(arr[i]);
}
```
> Array.from()  详细使用请点击
 [Array.from() MDN](https://developer.mozilla.org/zh-CN/docs/Web/JavaScript/Reference/Global_Objects/Array/from)
```
var arr=[1,2,3,4,5];
        
var arr2=Array.from(arr);
```
> 用扩展运算符复制数组
```
var arr=[1,2,3,4,5];
var arr2=[...arr];
```
